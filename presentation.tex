		top color=codebg,
		bottom color=codebg
	},

	arc=0.6mm,
	width= (\linewidth),
	fonttitle=\footnotesize,
	adjusted title=\texttt{#1},
	after title={%
		\hfill\setlength{\fboxsep}{2pt}

			{\footnotesize{\texttt{#2}}}}
	},

		\begin{tcbclipinterior}\fill[numberbg] (frame.south west)

	},

	right=0mm
}
